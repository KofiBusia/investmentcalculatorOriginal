Charts & Graphs — Fully Supported

Paste any of these directly in the article body. Chart.js and Plotly.js are pre-loaded on all article pages.

Chart.js Bar Chart

<canvas id="myChart" height="120"></canvas>
<script>
new Chart(document.getElementById('myChart'), {
  type: 'bar',
  data: {
    labels: ['Q1','Q2','Q3','Q4'],
    datasets: [{ label: 'Revenue', data: [120,180,150,220],
      backgroundColor: ['#4f46e5','#7c3aed','#10b981','#f59e0b'] }]
  }
});
</script>

Chart.js Line Chart

<canvas id="lineChart" height="100"></canvas>
<script>
new Chart(document.getElementById('lineChart'), {
  type: 'line',
  data: {
    labels: ['Jan','Feb','Mar','Apr','May'],
    datasets: [{ label: 'Stock Price', data: [10,14,12,17,20],
      borderColor:'#4f46e5', tension:0.3, fill:false }]
  }
});
</script>
